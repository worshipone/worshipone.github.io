1. 通过修改注册表的方式实现自启动

1.1 实验要求

编写代码，编辑注册表的Run/RunOnce/RunOnceEx键（任选其一，并明确三个键的区别），达到让某一程序在系统启动后自动运行的目的（可以以计算器、记事本等作为目标程序）。

1.2 实验环境

Windows 7或Windows 10主机（虚拟机）；

代码编辑器；

C/C++代码运行所需环境。

1.3 实验目的

了解恶意代码自启动常用手段。

1.4 实验过程
1.4.1 实验原理

  1. 打开注册表

    RegOpenKeyEx(HKEY_LOCAL_METHINE\SOFTWARE\Microsoft \Windows\CurrentVersion\Run)

  2. 添加键值

    键值为要启动程序的路径(可通过GetModuleFileName获得程序路径)

  3. 关闭注册表

    RegCloseKey

1.4.2 C语言代码
#include <windows.h>
#include <iostream>
#include <cstdlib>
using namespace std;

int main()
{
    HKEY hKey;
    LPCTSTR lpRun = "Software\\Microsoft\\Windows\\CurrentVersion\\Run";
    DWORD state,dwtype,sizeBuff;
    long lRet;
    char reBuff[100] = {0};


    //创建键
    lRet = RegCreateKeyEx(HKEY_CURRENT_USER,lpRun,0,NULL,0,0,NULL,&hKey,&state);
    if(lRet == ERROR_SUCCESS)
    {
        if(state == REG_CREATED_NEW_KEY)
            cout<<"表项创建成功"<<endl;
        RegCloseKey(hKey);
    }
    //创建或者修改键值
    lRet= RegOpenKeyEx(HKEY_CURRENT_USER, lpRun, 0, KEY_WRITE, &hKey);
    if(lRet == ERROR_SUCCESS)
    {
        RegSetValueEx(hKey, "QQ",0,REG_SZ,(BYTE *)"D:\\QQ\\Bin\\QQScLauncher.exe",100);
        RegCloseKey(hKey);
    }
    //读取键值
    lRet= RegOpenKeyEx(HKEY_CURRENT_USER, lpRun, 0, KEY_READ, &hKey);
    if(lRet == ERROR_SUCCESS)
    {
        sizeBuff = sizeof(reBuff);
        if(RegQueryValueEx(hKey,"QQ",0,&dwtype,(BYTE*)reBuff,&sizeBuff) == ERROR_SUCCESS){
            cout<<"读取QQ的键值"<<endl;
            cout<<reBuff<<endl;
        }
        RegCloseKey(hKey);
    }
    system("pause");
}

2. DLL文件编写

2.1 实验要求

编写一个DLL，使得在动态加载该DLL时，能够弹出“目标DLL已加载”的对话框。同时，为DLL添加两个导出函数，分别实现读取文件并打印出来，以及写入文件的功能，并且能够被其他程序动态调用。

2.2 实验环境

Windows 7或Windows 10主机（虚拟机）；

代码编辑器；

C/C++代码运行所需环境。

2.3 实验目的

了解DLL的作用和调用其函数的方法。

2.4 实验过程
2.4.1 实验原理

动态链接库是使用库来在多个应用程序之间共享代码的Windows特有方式； 一个DLL程序是不能独立运行的可执行文件，只能被EXE文件加载执行或者导出函数； 区别于静态链接库.LIB，帮助软件开发者和恶意代码作者最小化发布软件的大小规模。

2.4.2 Clion生成及使用dll

写在前面：dll项目负责生成dll，dllDemo项目负责引用dll

  1. dll项目所在CMakeLists.txt
cmake_minimum_required(VERSION 3.17)
project(dll)

set(CMAKE_CXX_STANDARD 14)
add_library(dll SHARED main.cpp)
  1. dllDemo项目的CMakeLists.txt
cmake_minimum_required(VERSION 3.17)
project(dllDemo)

set(CMAKE_CXX_STANDARD 14)
link_directories(lib)
add_executable(dllDemo main.cpp)
target_link_libraries(dllDemo yourDllName)

  1. dllDemo项目新建lib文件夹，并将dll项目生成的.dll文件放进去

  2. dllDemo项目的CMake Applocation添加环境变量，名称为PATH，值为lib路径

    //定义一个函数指针变量
    DLLWITHLIB pfFuncInDll = NULL;
    DLLWITHLIB pfReadFile = NULL;
    DLLWITHLIB pfWriteFile = NULL;
    //加载我们的dll
    HINSTANCE hinst=::LoadLibrary("libdll.dll");

    if (NULL != hinst)
    {
        std::cout<<"dll loaded!"<<std::endl;
    }
    //找到dll的FuncInDll函数
    pfFuncInDll = (DLLWITHLIB)GetProcAddress(hinst, "FuncInDll");
    pfReadFile = (DLLWITHLIB)GetProcAddress(hinst, "readFile");
    pfWriteFile = (DLLWITHLIB)GetProcAddress(hinst, "writeFile");
    //调用dll里的函数
    if (NULL != pfFuncInDll)
    {
        std::cout<<"**********\nTest function!"<<std::endl;
        (*pfFuncInDll)();
    }
    if (NULL != pfReadFile)
    {
        std::cout<<"**********\nRead file and print it!"<<std::endl;
        (*pfReadFile)();
    }
    if (NULL != pfWriteFile)
    {
        std::cout<<"**********\nWrite file!"<<std::endl;
        (*pfWriteFile)();
    }
    return 0;
}
